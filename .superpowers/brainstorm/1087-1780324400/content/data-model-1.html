扩展 GameState + Country 设计，附初始全局配置

📦 GameState — 全局状态

interface GameState {
  // === 基础 ===
  currentEra: 'POST_WW2' | 'IRON_CURTAIN' | 'INFO_AGE'
  year: number           // 1945-1991
  turn: number           // 1-184 (每季度1回合)
  phase: 'AI_EVENT' | 'INTEL' | 'PLAYER_ACTION' | 'RESOLVE'

  // === 冷战核心 ===
  globalTension: number  // 0-100 全球紧张度
  regionalScores: Record<string, RegionalScore>   // 区域级CWS
  countries: Record<string, Country>               // 国家级数据

  // === 双玩家 ===
  players: {
    usa: PlayerState
    ussr: PlayerState
  }

  // === 历史记录 ===
  timeline: TimelineEvent[]    // 供AI读取上下文
  newsFeed: NewsItem[]         // 底部新闻条
  activeBuffs: Buff[]          // 全局Buff/Debuff

  // === 科技 ===
  techTrees: { usa: TechNode[]; ussr: TechNode[] }

  // === 元数据 ===
  seed?: number  // 用于AI随机种子
}

📦 Country — 国家数据

interface Country {
  id: string           // "USA", "USSR", "DEU_WEST", "DEU_EAST"
  name: string
  region: string       // "NORTH_AMERICA" | "EUROPE" | "ASIA" | "MIDDLE_EAST" | "AFRICA" | "LATIN_AMERICA" | "OCEANIA"
  territoryPolygons: string[]   // GeoJSON feature IDs

  // === 核心冷战分 (0-100) ===
  coldWarScore: number

  // === 国力 ===
  economy: {
    gdp: number           // 相对值
    industry: number      // 1-100 工业化水平
    resources: number     // 自然资源
    budget: number        // 可用年度预算
  }
  military: {
    army: number          // 陆军规模/质量
    navy: number
    airforce: number
    nuclear: boolean      // 是否拥核
    nuclearArsenal: number // 核弹头数量
  }
  society: {
    stability: number     // 1-100 社会稳定
    morale: number        // 战争支持度
    population: number    // 人口(百万)
  }

  // === 阵营倾向 (0-100 两端) ===
  influence: {
    usa: number      // 亲美程度
    ussr: number     // 亲苏程度
  }
  // influence.usa + influence.ussr <= 100，差值决定倾向
  // 如 usa:70, ussr:10 → 亲西方；usa:10, ussr:60 → 亲苏

  // === 状态 ===
  alignment: 'USA_ALLY' | 'USSR_ALLY' | 'NON_ALIGNED'
  government: string     // "DEMOCRACY" | "COMMUNIST" | "DICTATORSHIP" | "MONARCHY" ...
  isFlashpoint: boolean  // 是否为当前危机焦点
}

📦 PlayerState — 玩家状态

interface PlayerState {
  side: 'USA' | 'USSR'
  leader: string            // 当前领导人

  // === 资源 ===
  actionPoints: number      // 每回合6点
  budget: number            // 国家预算
  prestige: number          // 国际威望

  // === 国内 ===
  publicSupport: number     // 1-100 国内支持率
  doomsdayClock: number     // 1-100 末日时钟(民众多恐慌)

  // === 情报 ===
  intel: {
    ussrPenetration: number // 对对方情报渗透程度
    cryptanalysis: boolean  // 是否能解密对方通信
  }

  // === 历史行动记录 (供AI上下文) ===
  recentActions: PlayerAction[]
}
