📐 科技树数据结构

interface TechNode {
  id: string              // "NUCLEAR_FISSION"
  name: string            // "核裂变"
  description: string     // 历史背景描述
  era: TechEra            // 所属时代
  yearRequirement: number // 最早可研发年份 (eg 1942)
  cost: number            // 所需科研点数

  // === 树结构 ===
  prerequisites: string[] // 前置科技 ID
  exclusiveTo?: 'USA' | 'USSR' | 'SHARED'

  // === 效果 ===
  effects: TechEffect[]

  // === 分支系统 ===
  branch: TechBranch      // 所属分支
}

type TechBranch = 'NUCLEAR' | 'AEROSPACE' | 'ELECTRONICS'
  | 'INDUSTRY' | 'MILITARY' | 'COMPUTING' | 'BIOLOGY'

type TechEra = 'ERA0_ATOMIC' | 'ERA1_MISSILE' | 'ERA2_SPACE'
  | 'ERA3_COMPUTER' | 'ERA4_INFORMATION'

⛔ 跨时代惩罚公式

最终研发成本 = baseCost × 【1 + (yearRequirement − currentYear) × penaltyFactor】
  • penaltyFactor = 0.5 — 每早 1 年成本增加 50%
  • 例：洲际导弹 (年需 1957)，在 1950 年研发 → 成本 × (1 + 7×0.5) = ×4.5
  • 提前 ≥10 年：成本 × (1 + 10×0.5) = ×6（上限封顶）
  • 绝对禁止：若 yearRequirement − currentYear > 15，AI 直接驳回"技术尚不存在"

    🧮 CWS 冷战分计算公式（核心引擎）

    CWS = 区域热点基准 + 大国博弈烈度 + 事件冲击

    CWS国家(n) = BASEn + INFLUENCE_GAPn + MILITARYn + EVENT_BOOSTn
    BASE = 该国历史基准热度 (0-30) — 德国东德25, 朝鲜20, 古巴10
    INFLUENCE_GAP = min( |usa−ussr| × 0.3, 30) — 双方争夺越激烈，热度越高
    MILITARY = (foreignTroops × 0.5 + militaryAid × 0.3) — 驻军和军援提升热度
    EVENT_BOOST = AI 裁判根据事件动态调整 (-20 ~ +30) — 危机事件临时飙升
    🔥 阈值规则：
    CWS > 60 → 区域不稳定，偶发低烈度冲突
    CWS > 75 → 高频冲突，代理人战争概率高
    CWS > 85 → 极易触发历史级危机/全面战争
    CWS > 95 → 核警戒状态，局势濒临失控

    🎛️ 区域CWS聚合

    RegionalCWSregion = AVG( CWScountries in region ) + 0.2 × MAX( CWScountries in region )
    区域内国家平均 CWS + 最高 CWS 的 20% 加成（热点国家会拉高整个区域紧张度）
